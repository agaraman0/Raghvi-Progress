#include<stdio.h>
int main()
{
    int n ,num,digit,rev=0;
    printf("enter the number: ");
    scanf("%d",&num);
    n=num;
do
{
    digit =num%10;
    rev=(rev*10)+digit;
    num=num/10;
}while(num!=0);
printf("reverse of number is: ",rev);
if(n==rev)
printf("number is palindrome");
else
printf("number is not palindrome")
}
